70 if (await this.templateService.FindTemplateByIdAsync(resumeIM.TemplateId) is null)

321

323
324 var emailRequest = new SendEmailRequest(share.Email, "Check out my resume", "Hi,\r\nCheck out my resume!", "resume.pdf", Convert.ToBase64String(fileMemoryStream.ToArray()));

326 await this.emailService.SendEmailAsync(emailRequest);
