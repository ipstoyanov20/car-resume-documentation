59 [HttpPost]
60 [Route("login")]
61 public async Task<IActionResult> LoginAsync([FromBody] UserLM userLM)
62 {
63 this.logger.LogInformation("User with email {Email} is trying to login.", userLM.Email);
64
65 if (!await this.authService.CheckIfUserExistsAsync(userLM.Email))
66 {
67 this.logger.LogWarning("User with email {Email} does not exist.", userLM.Email);
68 return this.BadRequest(new Response { Status = "login-failed", Message = "Invalid Email." });
69 }
70
71 if (!await this.authService.CheckIfUserHasVerifiedEmailAsync(userLM.Email))
72 {
73 this.logger.LogWarning("User with email {Email} has not verified their email.", userLM.Email);
74 return this.BadRequest(new Response { Status = "login-fielded", Message = "Email isn't verified." });
75 }
76
77 if (!await this.authService.CheckIsPasswordCorrectAsync(userLM.Email, userLM.Password))
78 {
79 this.logger.LogWarning("User with email {Email} has entered an incorrect password.", userLM.Email);
80 return this.BadRequest(new Response { Status = "login-failed", Message = "Invalid Password." });
81 }
82
83 var tokens = await this.tokenService.CreateTokensForUserAsync(userLM.Email);
84
85 this.logger.LogInformation("User with email {Email} has logged in successfully.", userLM.Email);
86
87 return this.Ok(new
88 {
89 AccessToken = new JwtSecurityTokenHandler().WriteToken(tokens.AccessToken),
90 RefreshToken = new JwtSecurityTokenHandler().WriteToken(tokens.RefreshToken),
91 Expiration = tokens.AccessToken!.ValidTo,
92 });
93 }

100 [HttpPost]
101 [Route("register")]
102 public async Task<IActionResult> RegisterAsync([FromBody] UserIM userIM)
103 {
104 this.logger.LogInformation("User with email {Email} is trying to register.", userIM.Email);
105
106 if (await this.authService.CheckIfUserExistsAsync(userIM.Email))
107 {
108 this.logger.LogWarning("User with email {Email} already exists.", userIM.Email);
109 return this.Conflict(new Response { Status = "register-failed", Message = "User already exists." });
110 }
111
112 var response = await this.authService.CreateUserAsync(userIM);
113
114 if (!response.Item1)
115 {
116 this.logger.LogWarning("User with email {Email} failed to register. Reason: {Reason}", userIM.Email, response.Item2);
117 return this.BadRequest(new Response { Status = "register-failed", Message = response.Item2 });
118 }
119
120 var token = await this.tokenService.GenerateEmailConfirmationTokenAsync(userIM.Email);
121 var callbackUrl = this.Url.Action("ConfirmEmail", "UserAuth", new { email = userIM.Email, token }, this.Request.Scheme, this.Request.Host.Value);
122
123 await this.emailService.SendEmailAsync(new (
124 userIM.Email!,
125 "Confirm your email",
126 $"Please confirm your account by <a href='{callbackUrl}'>clicking here</a>."));
127
128 this.logger.LogInformation("User with email {Email} has registered successfully.", userIM.Email);
129
130 return this.Ok(new Response { Status = "Success", Message = "User created successfully!" });
131 }

138 [HttpPost]
139 [Route("renew")]
140 public async Task<IActionResult> RenewAsync([FromBody] TokensIM tokensIM)
141 {
142 if (tokensIM is null)
143 {
144 return this.Unauthorized(new Response { Status = "renew-failed", Message = "No token provided." });
145 }
146
147 var tokens = await this.tokenService.CreateNewTokensAsync(tokensIM);
148
149 if (tokens.AccessToken is null)
150 {
151 return this.BadRequest(new Response { Status = "renew-failed", Message = "Invalid token." });
152 }
153
154 return new ObjectResult(new
155 {
156 accessToken = new JwtSecurityTokenHandler().WriteToken(tokens.AccessToken),
157 refreshToken = new JwtSecurityTokenHandler().WriteToken(tokens.RefreshToken),
158 Expiration = tokens.AccessToken!.ValidTo,
159 });
160 }
