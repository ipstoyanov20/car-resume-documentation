163 private bool ValidateRefreshToken(string token)
164 {

168 ValidateIssuer = false,

171 ValidateLifetime = false,
172 };

174 var tokenHandler = new JwtSecurityTokenHandler();

213 private JwtSecurityToken CreateToken(List<Claim> authClaims, TokenTypes tokenType)
214 {
215 SymmetricSecurityKey? authSigningKey;
216
217 int tokenValidity = 0;
218
219 if (tokenType == TokenTypes.AccessToken)
220 {
221 authSigningKey = new SymmetricSecurityKey(Encoding.UTF8.GetBytes(this.configuration["JWT:Secret"] !));
222 _ = int.TryParse(this.configuration["JWT:AccessTokenValidityInMinutes"], out tokenValidity);
223 }
224 else
225 {
226 authSigningKey = new SymmetricSecurityKey(Encoding.UTF8.GetBytes(this.configuration["JWT:RefreshTokenSecret"] !));
227 _ = int.TryParse(this.configuration["JWT:RefreshTokenValidityInDays"], out tokenValidity);
228 }
229
230 var token = new JwtSecurityToken(
231 expires: tokenType == TokenTypes.AccessToken ? DateTime.UtcNow.AddMinutes(tokenValidity) : DateTime.UtcNow.AddDays(tokenValidity),
232 claims: authClaims,
233 signingCredentials: new SigningCredentials(authSigningKey, SecurityAlgorithms.HmacSha256));
234
235 return token;
