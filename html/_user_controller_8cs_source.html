73 [HttpPost("change-password")]
74 public async Task<IActionResult> ChangerUserPasswordAsync([FromBody] UpdatePasswordModel updatePasswordModel)
75 {
76 this.logger.LogInformation("Changing password for user with id {UserId}", this.currentUser.UserId);
77
78 var user = await this.userService.GetUserByIdAsync(this.currentUser.UserId);
79
80 if (!await this.authService.CheckIsPasswordCorrectAsync(user!.Email, updatePasswordModel.OldPassword))
81 {
82 this.logger.LogWarning("Changing password for user with id {UserId} failed. Old password is incorrect.", this.currentUser.UserId);
83
84 return this.BadRequest(new Response
85 {
86 Status = "password-change-failed",
87 Message = "Old password is incorrect",
88 });
89 }
90
91 var result = await this.userService.ChangePasswordAsync(this.currentUser.UserId, updatePasswordModel.NewPassword);
92
93 if (!result.Succeeded)
94 {
95 this.logger.LogWarning("Changing password for user with id {UserId} failed. {Errors}", this.currentUser.UserId, result.Errors.First().Description);
96
97 return this.BadRequest(new Response
98 {
99 Status = "password-change-failed",
100 Message = result.Errors.FirstOrDefault()?.Description,
101 });
102 }
103
104 this.logger.LogInformation("Changing password for user with id {UserId} succeeded.", this.currentUser.UserId);
105
106 return this.Ok(new Response
107 {
108 Status = "password-changed-success",
